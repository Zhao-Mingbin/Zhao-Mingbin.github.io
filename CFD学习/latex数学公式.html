def convert(text,*arg):
    print('start******************************\n')
    raw = text.split('\n')
    #print(raw)
    for i in range(2,len(raw)):
        every = raw[i].split('|')
        for num in arg:
            #print(num)
            format = every[num+1].strip()[1:-1] 
            if format == '':
                format = ' '
                every[num] = f"{format}"
            else:
                every[num] = f"{format}" #第一个参数
        #print(every)#第二个参数
        new_every="|".join(every)
        raw[i] = new_every
    new_raw = "\n".join(raw)
    return new_raw
